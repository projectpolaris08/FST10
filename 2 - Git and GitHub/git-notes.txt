# Git Version
git --Version

# Configuring Git
git config --global user.name
git config --global user.email
git config user.name
git config user.email

# Initialize Git Repository
git init

# Legend
U - Untracked
A - Added
M - Modified

# Git Three Stage
-Check the status
git status

-Adding file to the staging area
git add .

-Adding a specific modified file to the staging area
git add [filename]

-Create a commit
git commit -m "message"

-Show the history
git log

# Branching and Merging
-Create a Branching
git branch [branch-name]

-List of branches
git branch -a

-Switch to a different branch
git switch [branch-name]

-Merging a branch to the master branch
git merge [branch-name]

-Delete a branch
git branch -d [branch-name]

# Remote Repository

-Add a remote to the local Repository
git remote add origin [repository-url]

-Confirming connection to the remote the remote repository
git remote -v

-Initial Push
git push -u origin main

-Push commits to the remote the remote repository
git push
